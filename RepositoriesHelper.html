def gpg_keys_edit
  keys = {}

  GpgKey.readable(current_organization).each{ |key|
    keys[key.id] = key.name
  }

  keys[""] = ""
  keys["selected"] = @repository.gpg_key_id || ""
  return keys.to_json
end
