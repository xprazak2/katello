def logins(users, locale)
  I18n.locale = locale
  @email = users.first.email
  @users = users
  mail :to => @email, :subject => _("Katello Logins")
end

def password_reset(user, locale)
  I18n.locale = locale
  @user = user
  mail :to => user.email, :subject => _("Katello User '%s' Password Reset") % user.username
end
