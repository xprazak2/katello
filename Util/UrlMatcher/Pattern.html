def static_match?(path)
  self.parts.each_with_index do |part,i|
    return false unless part[0] == ?: || path.parts[i] == part
  end
  true
end

def variables

  a = []
  self.parts.each_with_index do |part,i|
    a << @match.parts[i] if part[0] == ?:
  end

  a
end
