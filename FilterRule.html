def self.class_for( content_type)
  case content_type
  when PACKAGE
    PackageRule
  when PACKAGE_GROUP
    PackageGroupRule
  when ERRATA
    ErratumRule
  else
    params = {:content_type => content_type, :content_types => CONTENT_TYPES.join(", ")}
    raise (_("Invalid content type '%{content_type}' provided. Content types can be one of %{content_types}") % params)
  end
end
